= Configuring Axis Titles

=== Purpose

[[CodeSnippet]]
== Code Snippet

The following code examples show how to customize the titles on the x-axis and y-axis in the {FinancialChartName} control.

ifdef::jquery[]
*In JavaScript:*
[source,javascript]
----
$(".selector").{FinancialChartName}({
    dataSource: data,
    xAxisTitleMargin: 10,
    xAxisTitle: "Historical Date",
    xAxisTitleTextStyle: "12pt Arial",
    yAxisTitleTextStyle: "14pt Arial",
    yAxisTitle: "Share Price",
    yAxisTitleAngle: 90,
    yAxisTitleTextColor: "Green"
});
----
endif::jquery[]

ifdef::wpf[]
*In XAML:*
[source, xaml]
----
<Grid x:Name="layoutRoot">
    <ig:{FinancialChartName} ItemsSource="{Binding}"
        XAxisTitleFontSize="12"
        XAxisTitleMargin="10"
        XAxisTitle="Historical Date"
        XAxisTitleFontStyle="Italic"
        YAxisTitleFontSize="14"
        YAxisTitle="Share Price"
        YAxisTitleAngle="90"
        YAxisTitleForeground="Green" />
</Grid>
----
endif::wpf[]

ifdef::wpf[]
*In C#:*
[source, csharp]
----
var chart = new {FinancialChartName}();
chart.ItemsSource = data;
chart.XAxisTitleFontSize = 12;
chart.XAxisTitleMargin = new Thickness(10);
chart.XAxisTitle = "Historical Date";
chart.XAxisTitleFontStyle = FontStyles.Italic;
chart.YAxisTitleFontSize = 14;
chart.YAxisTitle = "Share Price";
chart.YAxisTitleAngle = 90;
chart.YAxisTitleForeground = Brushes.Green;

layoutRoot.Children.Add(chart);
----
endif::wpf[]

ifdef::wpf[]
*In Visual Basic*
[source, vb]
----
Dim chart = New {FinancialChartName}()
chart.ItemsSource = data
chart.XAxisTitleFontSize = 12
chart.XAxisTitleMargin = New Thickness(10)
chart.XAxisTitle = "Historical Date"
chart.XAxisTitleFontStyle = FontStyles.Italic
chart.YAxisTitleFontSize = 14
chart.YAxisTitle = "Share Price"
chart.YAxisTitleAngle = 90
chart.YAxisTitleForeground = Brushes.Green

layoutRoot.Children.Add(chart)

----
endif::wpf[]
